Marker Tracker Examples

To use a marker tracker, create a MarkerTracker object. In the simplest case, just pass the marker you wish to track and the map to it.

var mt = new MarkerTracker(marker,map);

You may also specify a number of options to customize. These options are passed via a MarkerTrackerOptions object, which contains the following fields:

  • iconScale: Scales the icon size by this value, 0 = no icon.
  • padding: Specifies the padding between the arrow and the edge of the map.
  • color: Color of the arrow.
  • weight: Thickness of the lines that make up the arrows.
  • length: Length of the arrow.
  • opacity: Opacity of the arrow.
  • updateEvent: The GMap2 event name that triggers the arrows to update.
  • panEvent: The GMarker event name that triggers a quick zoom to the tracked marker.

The MarkerTrackerOptions object is an object literal, so you simply declare the object without a constructor: 

var map = new GMap2(document.getElementById("map"));
map.setCenter(new GLatLng(37.4419, -122.1419), 13);
var mtOptions = { padding: 30, colors: '#aaaaaa', iconScale: 0.3 };
var mt = new MarkerTracker(marker map, mtOptions);

Simple MarkerTracker Examples

The following examples show how you can easily create trackers for markers:

An example with one MarkerTracker:

function loadMap() {
    if (GBrowserIsCompatible()) {
        map = new GMap2(document.getElementById("map"));
        map.setCenter(new GLatLng(36, -115), 4 );        
		map.addControl(new GMapTypeControl());
		map.addControl(new GSmallZoomControl());
		
		var point = new GLatLng(30, -120);
		marker = new GMarker(point);
		map.addOverlay(marker);
		
		var mt = new MarkerTracker(marker,map);
	}
}

}

Customizing the MarkerTracker

There are quite a few options available for customizing how the MarkerTracker looks and behaves. Below is an example of how to customize the marker tracker to your needs. Notice how the GIcon used for the original GMarker is inherited by the MarkerTracker. 

function loadMap() {
    if (GBrowserIsCompatible()) {
        map = new GMap2(document.getElementById("map"));
        map.setCenter(new GLatLng(36.0078125, -115.95703125), 4);        
		map.addControl(new GMapTypeControl());
		map.addControl(new GSmallZoomControl());
		
		var point1 = new GLatLng(37.419799,-122.064972);
		marker1 = new GMarker(point1);
		map.addOverlay(marker1);
		
		
		var lepIcon = new GIcon(marker1.getIcon());
		lepIcon.image = 'Animated-Leprechaun-1.gif';
		lepIcon.iconSize = new GSize(38,50);
		var point2 = new GLatLng(53.41291,-8.24389);
		marker2 = new GMarker( point2, lepIcon );
		map.addOverlay(marker2);
		
		
		
		var barIcon = new GIcon(marker1.getIcon());
		barIcon.image = 'bar.png';
		barIcon.iconSize = new GSize(32,32);
		var point3 = new GLatLng(-33.861293,151.208096);
		marker3 = new GMarker(point3, barIcon);
		map.addOverlay(marker3);
		
		
		
		var snowIcon = new GIcon(marker1.getIcon());
		snowIcon.image = 'snowflake_simple.png';
		snowIcon.iconSize = new GSize(32,32);
		var point4 = new GLatLng(-54.252389,-69.279785);
		marker4 = new GMarker(point4, snowIcon);
		map.addOverlay(marker4);
		
		
		// setup the trackers w/ options
		var mt1 = new MarkerTracker(marker1,map);
		
		opt2 = { 'color':'#00A300', 'weight':9, 'length': 20 };
		var mt2 = new MarkerTracker(marker2,map,opt2);	
		
		opt3 = { 'color':'#E0E000', 'weight':5, 'length': 10, 'padding':50, 'panEvent':'mouseover' };
		var mt3 = new MarkerTracker(marker3,map,opt3);
		
		opt4 = { 'color':'#00E0E0', 'weight':30, 'length': 40 };
		var mt4 = new MarkerTracker(marker4,map,opt4);
	}
}
